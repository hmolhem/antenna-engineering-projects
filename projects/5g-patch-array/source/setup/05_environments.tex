% --- Robust figure inclusion helper ---
% Allows this report to compile even when exported CST screenshots have not yet
% been copied into the expected figure folder. When the file exists, it is
% included normally; otherwise, a boxed placeholder is printed with the file path.
\newcommand{\maybeincludegraphics}[2][]{%
  \IfFileExists{#2}{%
    \includegraphics[#1]{#2}%
  }{%
    \fbox{%
      \begin{minipage}[c][0.22\textheight][c]{0.90\textwidth}
        \centering\small Missing figure file:\par\scriptsize\nolinkurl{#2}
      \end{minipage}%
    }%
  }%
}

% Centralized code-listing styles for MATLAB, Python,
% console output, Vitis HLS C/C++, Vitis output, and PowerShell.
%
% Required packages in setup/01_packages.tex:
%   \usepackage{xcolor}
%   \usepackage{listings}
%   \usepackage{textcomp}
%   \usepackage{upquote}   % optional; remove if unavailable
%
% Notes:
%   1. Each style name is defined only once.
%   2. Python colors are inspired by VS Code Light+ with a light-blue box.
%   3. Avoid redefining \lstdefinestyle{Python} elsewhere in the project.

% ---------------------------------------------------------
% MATLAB Listing Colors
% ---------------------------------------------------------
\definecolor{mGreen}{RGB}{34,139,34}
\definecolor{mGray}{RGB}{128,128,128}
\definecolor{mPurple}{RGB}{160,32,240}
\definecolor{mBackground}{RGB}{245,245,245}
\definecolor{mFrame}{RGB}{180,180,180}
\definecolor{mKeyword}{RGB}{0,0,180}

% ---------------------------------------------------------
% MATLAB Listing Style
% ---------------------------------------------------------
\lstdefinestyle{Matlab}{
    language=Matlab,
    backgroundcolor=\color{mBackground},
    basicstyle=\ttfamily\footnotesize,
    keywordstyle=\color{mKeyword}\bfseries,
    commentstyle=\color{mGreen}\itshape,
    stringstyle=\color{mPurple},
    numbers=left,
    numberstyle=\tiny\color{mGray},
    stepnumber=1,
    numbersep=8pt,
    frame=single,
    rulecolor=\color{mFrame},
    framerule=0.4pt,
    framesep=6pt,
    showspaces=false,
    showstringspaces=false,
    showtabs=false,
    keepspaces=true,
    columns=fullflexible,
    tabsize=4,
    breaklines=true,
    breakatwhitespace=true,
    captionpos=b,
    literate={≈}{$\approx$}{1}
}
